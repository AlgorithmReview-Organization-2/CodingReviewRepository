1번) 2660. Determine the Winner of a Bowling Game
시간 복잡도 : O(n)
아이디어 : index가 0일 때 1일 때를 그냥 if문으로 두고 continue로 처리함 가독성 좋게 하려고
자료구조 : 배열

--- 코드----

class Solution {
    public int isWinner(int[] player1, int[] player2) {
        int sum1 = 0;
        int sum2 = 0;

        for(int i = 0; i < player1.length ; i++){
            if(i == 0){
                sum1 += player1[i];
                continue;
            }
            
            if(i == 1){
                if(player1[i-1] == 10)
                    sum1 += player1[i]  * 2;
                else
                    sum1 += player1[i];
                continue;
            }

            if(player1[i-1] == 10 || player1[i-2] == 10)
                sum1 += player1[i] * 2;
            else
                sum1 += player1[i];
            
        }

        for(int i = 0; i < player2.length ; i++){
            if(i == 0){
                sum2 += player2[i];
                continue;
            }
            
            if(i == 1){
                if(player2[i-1] == 10)
                    sum2 += player2[i]  * 2;
                else
                    sum2 += player2[i];
                continue;
            }

            if(player2[i-1] == 10 || player2[i-2] == 10)
                sum2 += player2[i] * 2;
            else
                sum2 += player2[i];
            
        }

        
        return sum1 == sum2 ? 0 : sum1 > sum2 ? 1 : 2;
    }
}
